def plot_wave(time, amplitude, xtitle = 'Time (s)', \
              ytitle = 'Amplitude (arb.)', hold = False, \
              color = 'blue', marker = ',', legend = '', linestyle = '-'):
    ''' 
     time:      
         amplitude: 
         xtitle:     'Time (s)')
         ytitle:     'Amplitude (arb.)'
         hold:      FalseTrue 
         color:     
         marker:     pixel
         legend:    
         linestyle: 
    
    '''

    if (marker != ','):  #  pixel
        linestyle = ''   # 

    plt.plot(time, amplitude, color = color, marker = marker, \
             linestyle = linestyle, label = legend) # 

    if (legend != ''): # 
        plt.legend()

    if (hold == False): # hold == False plt.show() 
        plt.xlabel(xtitle)
        plt.ylabel(ytitle)
        plt.show()
